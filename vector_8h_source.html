8/*
9 A dynamic array type, in C!
10
11 NOTE: Unstable, so we do not gaurantee the items to remain at the same address during lifetime of the vector.
12
13 Usage:
14
15 // Create vector of integers
16 struct ye_vector* numbers = ye_new_vector(sizeof(int));
17
18 // Push some numbers
19 for(int i = 0; i < 5; i++) {
20 ye_vector_push_back(numbers, &i);
21 }
22
23 // Insert at front
24 int front_num = 42;
25 ye_vector_push_front(numbers, &front_num);
26
27 // Read and print values
28 for(size_t i = 0; i < numbers->size; i++) {
29 int* value = (int*)ye_vector_get(numbers, i);
30 printf("Index %zu: %d\n", i, *value);
31 }
32
33 // Modify value
34 int new_value = 100;
35 ye_vector_set(numbers, 1, &new_value);
36
37 // Remove element
38 ye_vector_remove(numbers, 2);
39
40 // Cleanup
41 ye_free_vector(numbers);
42*/
